/**
 * AreaChart
 * @alternateClassName AreaChart
 * @extends _Chart
 */
define('argos/Charts/AreaChart', [
    'dojo/_base/declare',
    'dojo/dom-attr',
    'dojo/dom-class',
    './_Chart',
    'dojox/charting/plot2d/Areas',
    'dojox/charting/plot2d/Markers',
    'dojox/charting/axis2d/Default'
], function(
    declare,
    domAttr,
    domClass,
    _Chart,
    Areas
) {
    return declare('argos.Charts.AreaChart', [_Chart], {
        xAxis: {
            fixLower: 'major',
            fixUpper: 'major',
            minorTicks: false
        },
        yAxis: {
            vertical: true,
            fixLower: 'major',
            fixUpper: 'major',
            minorTicks: false
        },
        plotType: Areas,
        plotOptions: {
            markers: false,
            tension: 'X'
        }
    });
});
